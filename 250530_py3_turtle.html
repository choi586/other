def sierpinski(length, depth):
    if depth == 0:
        for _ in range(3):
            forward(length)
            left(120)
    else:
        #    
        half = length / 2

        #  
        sierpinski(half, depth - 1)

        #  ,  
        forward(half)
        sierpinski(half, depth - 1)

        #   ,  
        backward(half)
        left(60)
        forward(half)
        right(60)

        #  
        sierpinski(half, depth - 1)

        #   
        left(60)
        backward(half)
        right(60)

def koch(length, depth):
    if depth == 0:
        forward(length)
    else:
        length /= 3
        koch(length, depth - 1)
        left(60)
        koch(length, depth - 1)
        right(120)
        koch(length, depth - 1)
        left(60)
        koch(length, depth - 1)

#   ( )
def koch_snowflake(length, depth):
    for _ in range(3):
        koch(length, depth)
        right(120)
